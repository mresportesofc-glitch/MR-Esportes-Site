🗄️ Tabelas Supabase

Execute os SQLs abaixo no seu projeto Supabase para criar as tabelas necessárias.

-- 1. Catálogo de camisas create table catalogo ( id bigint primary key default extract(epoch from now())*1000 + floor(random()*9999), user_id uuid references auth.users not null, nome text not null, categoria text, descricao text, preco_compra numeric default 0, preco_venda numeric default 0, criado_em timestamptz default now() ); alter table catalogo enable row level security; create policy "own" on catalogo using (auth.uid()=user_id); -- 2. Comandas create table comandas ( id bigint primary key default extract(epoch from now())*1000 + floor(random()*9999), user_id uuid references auth.users not null, cliente text not null, data_venda date, data_entrega date, frete numeric default 0, status text default 'Pendente', criado_em timestamptz default now() ); alter table comandas enable row level security; create policy "own" on comandas using (auth.uid()=user_id); -- 3. Itens das comandas create table comanda_itens ( id bigint primary key default extract(epoch from now())*1000 + floor(random()*9999), user_id uuid references auth.users not null, comanda_id bigint references comandas(id) on delete cascade, camisa_id bigint, camisa_nome text, camisa_categoria text, tamanho text, quantidade int default 1, preco_compra numeric default 0, preco_venda numeric default 0 ); alter table comanda_itens enable row level security; create policy "own" on comanda_itens using (auth.uid()=user_id);
